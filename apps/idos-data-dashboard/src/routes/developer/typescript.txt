// [ frontend ] idOS client + enclave

import { createIDOSClient, type idOSClient } from "@idos-network/client";

// `existingWalletSigner` = whatever your app already uses (e.g. ethers Signer).
declare const existingWalletSigner: { getAddress(): Promise<string> };

export async function createIdosClientForApp(): Promise<idOSClient> {
  return createIDOSClient({
    nodeUrl: publicConfig.IDOS_NETWORK_URL,
    enclaveOptions: { container: "#idOS-enclave" },
  });
}

export async function bindUserSigner(idosClient: idOSClient) {
  return idosClient.withUserSigner(existingWalletSigner as never);
}

export async function checkProfile(idosClient: idOSClient) {
  const address = await existingWalletSigner.getAddress();
  const hasProfile = await idosClient.addressHasProfile(address);
  return { address, hasProfile };
}

// After `withUserSigner`, use the bound client for credential/grant APIs.
import type { idOSClient as IdosClientType } from "@idos-network/client";

type IdosWithUser = Awaited<ReturnType<idOSClient["withUserSigner"]>>;

export async function filterAndGrantFlow(idosWithUser: IdosWithUser) {
  const credentials = await idosWithUser.filterCredentials({
    acceptedIssuers: [
      {
        authPublicKey: publicConfig.IDOS_ACCEPTED_ISSUER_PUBLIC_KEY,
      },
    ],
    publicNotesFieldFilters: {
      pick: {},
      omit: {},
    },
    privateFieldFilters: {
      pick: {},
      omit: {},
    },
  });

  const grants = await idosWithUser.getAccessGrantsOwned();
  const existingGrant = grants.find(
    (grant) =>
      grant.ag_grantee_wallet_identifier ===
      publicConfig.IDOS_CONSUMER_PUBLIC_IDENTIFIER,
  );

  return { credentials, grants, existingGrant };
}

declare const selectedCredential: { id: string };
declare const userWalletIdentifier: string;
declare const lockedUntilUnixTimestamp: number;

export async function requestGrantIfNeeded(idosWithUser: IdosWithUser) {
  const accessGrant = await idosWithUser.requestAccessGrant(selectedCredential.id);
  return accessGrant;
}

export async function requestDelegatedGrantMessage(idosWithUser: IdosWithUser) {
  const dag = await idosWithUser.requestDAGMessage({
    dag_owner_wallet_identifier: userWalletIdentifier,
    dag_grantee_wallet_identifier: publicConfig.IDOS_CONSUMER_PUBLIC_IDENTIFIER,
    dag_data_id: selectedCredential.id,
    dag_locked_until: lockedUntilUnixTimestamp,
  });
  return dag;
}

// [ backend ] KYC relay JWT → link (Express-style; adapt to your framework)
// =============================================================================

import type { Request, Response } from "express";
import jwt from "jsonwebtoken";

declare function requireAuth(
  req: Request,
  res: Response,
  next: () => void,
): void;

declare global {
  namespace Express {
    interface User {
      id: string;
      relayClientId: string;
      walletAddress: string;
    }
  }
}

export function registerKycRoute(app: import("express").Express) {
  app.get("/kyc", requireAuth, (req, res) => {
    const user = req.user!;

    const token = jwt.sign(
      {
        clientId: user.relayClientId,
        level: "plus+liveness+idos",
        walletAddress: user.walletAddress,
        externalUserId: user.id,
        kyc: true,
      },
      process.env.IDOS_RELAY_PRIVATE_KEY!,
      { algorithm: "ES512", expiresIn: "1h" },
    );

    res.json({
      link: `${process.env.IDOS_RELAY_URL}/kyc?token=${encodeURIComponent(token)}`,
    });
  });
}

// [ frontend ] KYC iframe + postMessage (React)
// =============================================================================

import { useCallback, useEffect, useState } from "react";

export function useKycIframeRelay() {
  const [kycLink, setKycLink] = useState<string | null>(null);
  const [message, setMessage] = useState<string | null>(null);

  const relayOrigin = new URL(publicConfig.IDOS_RELAY_URL).origin;

  const messageReceiver = useCallback(
    (ev: MessageEvent) => {
      if (ev.origin !== relayOrigin) return;

      if (ev.data?.error) {
        setMessage(`KYC process failed with: ${JSON.stringify(ev.data.error)}`);
        setKycLink(null);
        return;
      }

      if (ev.data?.open) {
        window.open(ev.data.open, ev.data.target, ev.data.features);
        return;
      }

      setMessage("KYC process is completed.");
      setKycLink(null);
    },
    [relayOrigin],
  );

  useEffect(() => {
    window.addEventListener("message", messageReceiver);
    return () => window.removeEventListener("message", messageReceiver);
  }, [messageReceiver]);

  return { kycLink, setKycLink, message };
}

// [ backend ] Consumer SDK init — Ed25519 / Nacl

import { idOSConsumer as idOSConsumerClass } from "@idos-network/consumer";
import { hexDecode } from "@idos-network/utils/codecs";
import nacl from "tweetnacl";

export async function initIdosConsumerNacl() {
  const secretKey = hexDecode(process.env.IDOS_CONSUMER_AUTH_KEY!);
  const consumerSigner = nacl.sign.keyPair.fromSecretKey(secretKey);

  return idOSConsumerClass.init({
    consumerSigner,
    recipientEncryptionPrivateKey: process.env.IDOS_CONSUMER_ENC_KEY!,
    nodeUrl: process.env.IDOS_NETWORK_URL,
  });
}

import { Wallet } from "ethers";

export async function initIdosConsumerEvm() {
  const consumerSigner = new Wallet(process.env.IDOS_CONSUMER_AUTH_KEY!);

  return idOSConsumerClass.init({
    consumerSigner,
    recipientEncryptionPrivateKey: process.env.IDOS_CONSUMER_ENC_KEY!,
    nodeUrl: process.env.IDOS_NETWORK_URL,
  });
}

// =============================================================================
// [ backend ] Credential verification
// =============================================================================

export async function verifyCredentialExample(credentialContents: string) {
  const allowedIssuers = [
    {
      issuer: process.env.IDOS_ACCEPTED_ISSUER_ID!,
      publicKeyMultibase: process.env.IDOS_ACCEPTED_ISSUER_PUBLIC_KEY!,
    },
  ];

  const verificationResults = await idosConsumer.verifyCredential(
    credentialContents,
    allowedIssuers,
  );

  const [verificationResult] = verificationResults;

  if (!verificationResult?.verified) {
    throw new Error("idOS credential verification failed");
  }

  return verificationResult;
}
